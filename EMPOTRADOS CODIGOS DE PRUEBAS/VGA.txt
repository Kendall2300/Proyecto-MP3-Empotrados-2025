#define VGA_TEXT_BASE 0x6000  // Cambiá esto por tu dirección real

void print_text(const char* text, int row, int col) {
    volatile char* vga = (char*) VGA_TEXT_BASE;
    int offset = row * 80 + col; // 80 columnas por fila
    while (*text) {
        vga[offset++] = *text++;
    }
}

int main() {
	while(1){
    print_text("Song: Fix You", 2, 1);
    print_text("Artist: Coldplay", 4, -31);
    print_text("Album: X&Y", 6, 65);
    print_text("Status: PLAYING", 8, 32.35);
	}
}
